  • Then we need to write the POST method code for function the page properly. 
    • //POST Method
        public async Task<IActionResult> OnPostAsync()
        {
            if (ModelState.IsValid)
            {
                var user = new IdentityUser()
                {
                    UserName = Model?.Email,
                    Email = Model?.Email
                };

                var result = await userManager.CreateAsync(user,Model?.Password);

                if (result.Succeeded)
                {
                    await signInManager.SignInAsync(user, false);
                    return RedirectToPage("Index");
                }

                foreach(var error in result.Errors)
                {
                    ModelState.AddModelError("", error.Description);
                }
            }
            return Page();

  • For Adding functionality of the Login form we need to write the POST method. So go to Login.cshtml.cs page and write the code below from OnGet().
  • First create a Constructor of LoginModel and pass the signInManager in the bracket.
  • Then click on SignInManager and press Ctrl + . and select the Create and SignInProperty(...), that automatically write the code. 
    • public LoginModel(SignInManager<IdentityUser> signInManager)
    {
        this.signInManager = signInManager;
     }
  • Then create a OnPostAsync() for POST method and write the code for that method. 
    • public async Task<IActionResult> OnPostAsync(string? returnUrl = null)
        {
            if (ModelState.IsValid)
            {
                var identityResult = await signInManager.PasswordSignInAsync(Model?.Email, Model?.Password, Model.RememberMe, false);
               if (identityResult.Succeeded)
               {
                    if (returnUrl == null || returnUrl == "/")
                    {
                        return RedirectToPage("Index");
                    }
                    else
                    {
                        return RedirectToPage(returnUrl);
                    }
               }
                ModelState.AddModelError("", "Username pr Password incorrect");
            }
            return Page();
        }
  • Then run the code for testing the page. If the page work properly and the page is redirect to the Index page with the correct ID and Password.
